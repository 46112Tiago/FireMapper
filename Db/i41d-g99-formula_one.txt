DRIVER:
*Driver_Id: Number
*Name: String
*Country: String
*Team: (Team_Id on TEAM)
*Position: String
*Poles: Number

GRAND_PRIX:
*Gp_Id: Number
*Location: String
*Winner: (Driver_Id on DRIVER)

TEAM:
*Team_Id: Number
*Name: String
*Base: String
*CEO: String
*First_Team_Entry: Number

CAR:
*Car_Id: Number
*Team: (Team_Id on TEAM)
*Driver: (Driver_Id on TEAM)
*Engine_Id: String
